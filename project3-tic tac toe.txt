#include<iostream>
#include<stdlib.h>
using namespace std;
char board[3][3] = {{'1', '2','3'}, {'4','5','6'},{'7', '8', '9'}};
int choice;
int row, column;
char turn = 'X';
bool draw = false;
void boardDisplay(){
cout << "\n\n\tPLAYER 1 - X \tPLAYER 2 - O\n\n"<<
        "\t\t\t   |   |   "<<
        "\n\t\t\t "<< board[0][0]<<" | "<< board[0][1] << " | "<< board[0][2]<<
        "\n\t\t\t___|___|___\n"<<
        "\t\t\t   |   |   "<<
        "\n\t\t\t "<< board[1][0]<<" | "<< board[1][1] << " | "<< board[1][2]<<
        "\n\t\t\t___|___|___\n"<<
        "\t\t\t   |   |   "<<
        "\n\t\t\t "<< board[2][0]<<" | "<< board[2][1] << " | "<< board[2][2]<<
        "\n\t\t\t   |   |   \n\n";
}
void player_turn(){
if(turn == 'X'){
    cout << "\n\tPlayer 1 - [x] turn: ";
}else if(turn == 'O'){
    cout << "\n\tPLayer 2 - [O] turn: ";
}
cin >> choice;
switch(choice){
    case 1: row =0; column = 0;break;
    case 2: row =0; column = 1;break;
    case 3: row =0; column = 2;break;
    case 4: row =1; column = 0;break;
    case 5: row =1; column = 1;break;
    case 6: row =1; column = 2;break;
    case 7: row =2; column = 0;break;
    case 8: row =2; column = 1;break;
    case 9: row =2; column = 2;break;
    default: cout << "\n\t\tINVALID MOVE!!";
}
if(turn == 'X'&& board[row][column] != 'X' && board[row][column]!='O'){
    board[row][column] = 'X';
    turn = '0';
}else if(turn == 'O'&& board[row][column] != 'X' && board[row][column]!='O'){
    board[row][column] = '0';
    turn = 'X';
}else{
    cout << "Box is already filled!";
    player_turn();
}
boardDisplay();
}
bool gameover(){
    for(int i =0; i<3; i++){
        if(board[i][0]==board[i][1] && board[i][0]==board[i][2]){
        return false;
        }
    }
    for(int i =0; i<3; i++){
        if(board[0][i]==board[1][i] && board[0][i]==board[2][i]){
        return false;
        }
    }
    for(int i =0; i<3; i++){
        if(board[0][0]==board[1][1] && board[0][0]==board[2][2]){
        return false;
        }
    }
    for(int i =0; i<3; i++){
        if(board[0][2]==board[1][1] && board[0][2]==board[2][0]){
        return false;
        }
    }

    for(int i = 0; i<3; i++){
        for(int j = 0; j < 3; j++){
            if(board[i][j] != 'X' && board[i][j] != 'O'){
                return true;
            }
        }
    }
    draw = true;
    return false;
}
int main(){
    cout << "\t\t***TIC-TAC-TOE-GAME***";
    cout << "\n\t\t-->For 2 players.";

    while(gameover){
        boardDisplay();
        player_turn();
        gameover();
    }
    if(turn == 'X' && draw == false){
        cout << "\n\nCongratulation! Player with [X] has won the game";
    }else if(turn == 'O' && draw == false){
        cout << "\n\nCongratulation! Player with [O] has won the game";
    }else{
        cout << "\n\nThe game is draw!!\n\n";
    }

    return 0;
}
